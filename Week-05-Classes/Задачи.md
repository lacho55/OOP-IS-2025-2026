<div class="task-block">
  <h2>Task 1</h2>
  <p>Да се напишат следните класове:</p>
  <ul>
    <li><strong>Учен</strong> — всеки учен има име, длъжност, заплата и колко пари печели за университета с труда си за един ден.</li>
    <li><strong>Университет</strong> — в него има учени.</li>
  </ul>
  <p><em>Академичните длъжности са: <strong>асистент, главен асистент, доцент, професор</strong></em></p>
  <p>Напишете функции, които да позволяват следните действия:</p>
  <ul>
    <li>Да се назначи учен.</li>
    <li>Да се уволни учен.</li>
    <li>Учен да се повиши (сменя си длъжността и заплатата; <em>разликата в заплатите между всички длъжности е 100 лв.</em>)</li>
    <li>Да се намалят заплатите на всички учени на дадена длъжност.</li>
    <li>Да се съкратят всички учени на дадена длъжност.</li>
    <li>Да се сметне печалбата (<em>profit</em>) за един месец (приход (<em>income</em>) минус разходи (<em>expenses</em>)).</li>
    <li>Да се изведат на конзолата всички служители и печалбата.</li>
  </ul>

  <details>
    <summary>Решение</summary>
<pre>#include &lt;iostream&gt;
#include &lt;cstring&gt;

const int MAX_SCIENTISTS = 100;
const int WORK_DAYS      = 20;   // работни дни в месец

// Длъжности като константи
const int ASSISTANT   = 0;
const int SENIOR_ASST = 1;
const int DOCENT      = 2;
const int PROFESSOR   = 3;

const char*  TITLES[]     = { "Асистент", "Главен асистент", "Доцент", "Професор" };
const double BASE_SALARY  = 1000.0;
const double SALARY_STEP  =  100.0;

struct Scientist {
    char   name[64];
    int    rank;         // 0=асистент .. 3=професор
    double salary;
    double dailyIncome;  // пари, носени на университета за 1 ден
};

struct University {
    Scientist scientists[MAX_SCIENTISTS];
    int       count;

    University() : count(0) {}

    // Назначава нов учен
    void hire(const char* name, int rank, double dailyIncome) {
        if (count >= MAX_SCIENTISTS) {
            std::cout &lt;&lt; "Няма свободни места!" &lt;&lt; std::endl;
            return;
        }
        strncpy(scientists[count].name, name, 63);
        scientists[count].rank        = rank;
        scientists[count].salary      = BASE_SALARY + rank * SALARY_STEP;
        scientists[count].dailyIncome = dailyIncome;
        count++;
        std::cout &lt;&lt; name &lt;&lt; " е назначен." &lt;&lt; std::endl;
    }

    // Уволнява учен по индекс (запълва дупката с последния елемент)
    void fire(int index) {
        if (index &lt; 0 || index &gt;= count) return;
        std::cout &lt;&lt; scientists[index].name &lt;&lt; " е уволнен." &lt;&lt; std::endl;
        scientists[index] = scientists[count - 1];
        count--;
    }

    // Уволнява учен по ime
    void fire(const char* name) {
        for (int i = 0; i &lt; count; i++) {
            if (strcmp(scientists[i].name, name) == 0) { fire(i); return; }
        }
        std::cout &lt;&lt; "Учен с това ime не е намерен." &lt;&lt; std::endl;
    }

    // Повишава учен по ime
    void promote(const char* name) {
        for (int i = 0; i &lt; count; i++) {
            if (strcmp(scientists[i].name, name) == 0) {
                if (scientists[i].rank == PROFESSOR) {
                    std::cout &lt;&lt; scientists[i].name &lt;&lt; " вече е Професор." &lt;&lt; std::endl;
                    return;
                }
                scientists[i].rank++;
                scientists[i].salary += SALARY_STEP;
                std::cout &lt;&lt; scientists[i].name &lt;&lt; " е повишен до "
                          &lt;&lt; TITLES[scientists[i].rank] &lt;&lt; std::endl;
                return;
            }
        }
        std::cout &lt;&lt; "Учен с това ime не е намерен." &lt;&lt; std::endl;
    }

    // Намалява заплатите на всички учени с дадена длъжност
    void reduceSalaries(int rank, double amount) {
        for (int i = 0; i &lt; count; i++) {
            if (scientists[i].rank == rank)
                scientists[i].salary -= amount;
        }
        std::cout &lt;&lt; "Заплатите на " &lt;&lt; TITLES[rank]
                  &lt;&lt; " са намалени с " &lt;&lt; amount &lt;&lt; " лв." &lt;&lt; std::endl;
    }

    // Съкращава всички учени с дадена длъжност
    void fireByRank(int rank) {
        for (int i = count - 1; i &gt;= 0; i--) {
            if (scientists[i].rank == rank) fire(i);
        }
    }

    // Месечна печалба
    double monthlyProfit() const {
        double income = 0, expenses = 0;
        for (int i = 0; i &lt; count; i++) {
            income   += scientists[i].dailyIncome * WORK_DAYS;
            expenses += scientists[i].salary;
        }
        return income - expenses;
    }

    // Извежда всички служители и печалбата
    void print() const {
        std::cout &lt;&lt; "\n=== Служители ===" &lt;&lt; std::endl;
        double income = 0, expenses = 0;
        for (int i = 0; i &lt; count; i++) {
            std::cout &lt;&lt; "  " &lt;&lt; scientists[i].name
                      &lt;&lt; " | " &lt;&lt; TITLES[scientists[i].rank]
                      &lt;&lt; " | Заплата: " &lt;&lt; scientists[i].salary
                      &lt;&lt; " | Дневен приход: " &lt;&lt; scientists[i].dailyIncome
                      &lt;&lt; std::endl;
            income   += scientists[i].dailyIncome * WORK_DAYS;
            expenses += scientists[i].salary;
        }
        std::cout &lt;&lt; "Приход:  " &lt;&lt; income            &lt;&lt; " лв.\n"
                  &lt;&lt; "Разходи: " &lt;&lt; expenses          &lt;&lt; " лв.\n"
                  &lt;&lt; "Печалба: " &lt;&lt; (income-expenses) &lt;&lt; " лв." &lt;&lt; std::endl;
    }
};

int main() {
    University uni;
    uni.hire("Иван Иванов",   ASSISTANT, 80);
    uni.hire("Мария Петрова", DOCENT,   150);
    uni.hire("Георги Колев",  PROFESSOR,200);
    uni.hire("Петър Стоянов", ASSISTANT, 75);

    uni.promote("Иван Иванов");
    uni.reduceSalaries(ASSISTANT, 50);
    uni.print();

    uni.fireByRank(SENIOR_ASST);
    uni.print();
    return 0;
}</pre>
    <div class="note">💡 Длъжностите са числа 0–3, така повишението е просто <code>rank++</code> и <code>salary += 100</code>. При уволнение запълваме дупката с последния елемент, за да не местим целия масив.</div>

<div class="task-block">
  <h2>Task 2</h2>
  <p>Да се напишат следните класове:</p>
  <ul>
    <li><strong>Вафла</strong> — марка, грамаж, цена на едро, цена за краен клиент.</li>
    <li><strong>Магазин за вафли</strong> — масив от вафли, максимален брой вафли, приходи до момента, разходи до момента.</li>
  </ul>
  <p>Напишете функции, които да позволяват следните действия:</p>
  <ul>
    <li>Продава се вафла.</li>
    <li>Поръчват се вафли от склада на дадена марка.</li>
    <li>Показва колко вафли има от дадена марка или даден грамаж.</li>
    <li>Показва печалбата до момента.</li>
  </ul>

  <details>
    <summary>Решение</summary>
<pre>#include &lt;iostream&gt;
#include &lt;cstring&gt;

const int MAX_TYPES = 50;

struct Wafer {
    char   brand[64];
    int    grams;
    double wholesalePrice;  // покупна цена
    double retailPrice;     // цена за клиент
    int    quantity;
};

struct WaferShop {
    Wafer  stock[MAX_TYPES];
    int    typeCount;
    double income;
    double expenses;

    WaferShop() : typeCount(0), income(0), expenses(0) {}

    // Помощна: търси индекс по марка
    int findByBrand(const char* brand) const {
        for (int i = 0; i &lt; typeCount; i++) {
            if (strcmp(stock[i].brand, brand) == 0) return i;
        }
        return -1;
    }

    // Поръчва вафли (добавя нов вид или увеличава количеството)
    void order(const char* brand, int grams,
               double wholesale, double retail, int qty) {
        int idx = findByBrand(brand);
        if (idx == -1) {
            if (typeCount &gt;= MAX_TYPES) {
                std::cout &lt;&lt; "Складът е пълен!" &lt;&lt; std::endl;
                return;
            }
            strncpy(stock[typeCount].brand, brand, 63);
            stock[typeCount].grams          = grams;
            stock[typeCount].wholesalePrice = wholesale;
            stock[typeCount].retailPrice    = retail;
            stock[typeCount].quantity       = qty;
            typeCount++;
        } else {
            stock[idx].quantity += qty;
        }
        expenses += wholesale * qty;
        std::cout &lt;&lt; "Поръчани " &lt;&lt; qty &lt;&lt; " бр. " &lt;&lt; brand &lt;&lt; std::endl;
    }

    // Продава една вафла от дадена марка
    void sell(const char* brand) {
        int idx = findByBrand(brand);
        if (idx == -1 || stock[idx].quantity == 0) {
            std::cout &lt;&lt; "Няма налични вафли от марка " &lt;&lt; brand &lt;&lt; std::endl;
            return;
        }
        stock[idx].quantity--;
        income += stock[idx].retailPrice;
        std::cout &lt;&lt; "Продадена " &lt;&lt; brand
                  &lt;&lt; " за " &lt;&lt; stock[idx].retailPrice &lt;&lt; " лв." &lt;&lt; std::endl;
    }

    // Брой вафли от дадена марка
    void countByBrand(const char* brand) const {
        int idx = findByBrand(brand);
        if (idx == -1) {
            std::cout &lt;&lt; "Марката " &lt;&lt; brand &lt;&lt; " не е намерена." &lt;&lt; std::endl;
            return;
        }
        std::cout &lt;&lt; "Марка " &lt;&lt; brand &lt;&lt; ": " &lt;&lt; stock[idx].quantity &lt;&lt; " бр." &lt;&lt; std::endl;
    }

    // Брой вафли от даден грамаж
    void countByGrams(int grams) const {
        int total = 0;
        for (int i = 0; i &lt; typeCount; i++) {
            if (stock[i].grams == grams) total += stock[i].quantity;
        }
        std::cout &lt;&lt; "Вафли " &lt;&lt; grams &lt;&lt; "г: " &lt;&lt; total &lt;&lt; " бр." &lt;&lt; std::endl;
    }

    // Печалба до момента
    void printProfit() const {
        std::cout &lt;&lt; "Приходи:  " &lt;&lt; income            &lt;&lt; " лв.\n"
                  &lt;&lt; "Разходи:  " &lt;&lt; expenses          &lt;&lt; " лв.\n"
                  &lt;&lt; "Печалба:  " &lt;&lt; (income-expenses) &lt;&lt; " лв." &lt;&lt; std::endl;
    }
};

int main() {
    WaferShop shop;
    shop.order("Kinder", 21, 0.80, 1.20, 50);
    shop.order("Balsen", 35, 0.60, 1.00, 30);
    shop.order("Kinder", 21, 0.80, 1.20, 20);  // дозареждане

    shop.sell("Kinder");
    shop.sell("Balsen");
    shop.sell("Kinder");

    shop.countByBrand("Kinder");
    shop.countByGrams(21);
    shop.printProfit();
    return 0;
}</pre>
    <div class="note">💡 При повторна поръчка на същата марка само увеличаваме <code>quantity</code>, без да добавяме нов запис. Разходите се отчитат при поръчка, приходите — при продажба.</div>
  </details>

<div class="task-block">
  <h2>Task 3</h2>
  <p>Да се напишат следните класове:</p>
  <ul>
    <li><strong>Студент</strong> — имена, факултетен номер, специалност и общ брой получени кредити.</li>
    <li><strong>Специалност</strong> — съдържа всички студенти, записани в нея.</li>
    <li><strong>Предмет</strong> — заглавие, преподавател, специалност и брой кредити.</li>
    <li><strong>Университет</strong> — специалности и предмети.</li>
  </ul>
  <p>Напишете функции, които да позволяват следните действия:</p>
  <ul>
    <li>Студент да бъде приет в дадена специалност.</li>
    <li>Една специалност да "изкара" даден предмет.</li>
    <li>Една специалност да "изкара" даден предмет, като произволни студенти са "скъсани".</li>
    <li>Студент да може да премине от една в друга специалност, стига да има натрупани 30 кредита.</li>
    <li>Студент да завърши — автоматично след 180 кредита.</li>
    <li>Да се изведе на конзолата съдържанието на университета.</li>
  </ul>

  <details>
    <summary>Решение</summary>
<pre>#include &lt;iostream&gt;
#include &lt;cstring&gt;
#include &lt;cstdlib&gt;
#include &lt;ctime&gt;

const int MAX_STUDENTS   = 200;
const int MAX_SPECIALTIES =  20;
const int MAX_SUBJECTS   =  50;
const int CREDITS_TRANSFER  =  30;
const int CREDITS_GRADUATE  = 180;

struct Specialty;  // предварителна декларация

// ── Студент ───────────────────────────────────────────────────────
struct Student {
    char      name[64];
    int       fn;
    int       credits;
    bool      graduated;
    Specialty* specialty;

    Student() : fn(0), credits(0), graduated(false), specialty(nullptr) {
        name[0] = '\0';
    }
};

// ── Специалност ───────────────────────────────────────────────────
struct Specialty {
    char     name[64];
    Student* students[MAX_STUDENTS];
    int      studentCount;

    Specialty() : studentCount(0) { name[0] = '\0'; }

    bool addStudent(Student* s) {
        if (studentCount &gt;= MAX_STUDENTS) return false;
        students[studentCount++] = s;
        s-&gt;specialty = this;
        return true;
    }

    void removeStudent(Student* s) {
        for (int i = 0; i &lt; studentCount; i++) {
            if (students[i] == s) {
                students[i] = students[studentCount - 1];
                studentCount--;
                return;
            }
        }
    }
};

// ── Предмет ───────────────────────────────────────────────────────
struct Subject {
    char      title[64];
    char      teacher[64];
    Specialty* specialty;
    int        credits;

    Subject() : specialty(nullptr), credits(0) {
        title[0] = teacher[0] = '\0';
    }
};

// ── Университет ───────────────────────────────────────────────────
struct University {
    Student   students[MAX_STUDENTS];
    int       studentCount;

    Specialty specialties[MAX_SPECIALTIES];
    int       specialtyCount;

    Subject   subjects[MAX_SUBJECTS];
    int       subjectCount;

    University() : studentCount(0), specialtyCount(0), subjectCount(0) {}

    Specialty* addSpecialty(const char* name) {
        if (specialtyCount &gt;= MAX_SPECIALTIES) return nullptr;
        strncpy(specialties[specialtyCount].name, name, 63);
        return &amp;specialties[specialtyCount++];
    }

    Subject* addSubject(const char* title, const char* teacher,
                        Specialty* spec, int credits) {
        if (subjectCount &gt;= MAX_SUBJECTS) return nullptr;
        strncpy(subjects[subjectCount].title,   title,   63);
        strncpy(subjects[subjectCount].teacher, teacher, 63);
        subjects[subjectCount].specialty = spec;
        subjects[subjectCount].credits   = credits;
        return &amp;subjects[subjectCount++];
    }

    // Приема студент в специалност
    Student* enroll(const char* name, int fn, Specialty* spec) {
        if (studentCount &gt;= MAX_STUDENTS) return nullptr;
        strncpy(students[studentCount].name, name, 63);
        students[studentCount].fn      = fn;
        students[studentCount].credits = 0;
        students[studentCount].graduated = false;
        spec-&gt;addStudent(&amp;students[studentCount]);
        std::cout &lt;&lt; name &lt;&lt; " е приет в " &lt;&lt; spec-&gt;name &lt;&lt; std::endl;
        return &amp;students[studentCount++];
    }

    // Проверява дали студентът трябва да завърши
    void checkGraduation(Student* s) {
        if (!s-&gt;graduated &amp;&amp; s-&gt;credits &gt;= CREDITS_GRADUATE) {
            s-&gt;graduated = true;
            s-&gt;specialty-&gt;removeStudent(s);
            std::cout &lt;&lt; "  [" &lt;&lt; s-&gt;name &lt;&lt; " ЗАВЪРШИ!]" &lt;&lt; std::endl;
        }
    }

    // Специалността изкарва предмет — всички студенти получават кредити
    void passSubject(Specialty* spec, Subject* subj) {
        if (spec != subj-&gt;specialty) {
            std::cout &lt;&lt; "Предметът не е за тази специалност!" &lt;&lt; std::endl;
            return;
        }
        std::cout &lt;&lt; spec-&gt;name &lt;&lt; " изкарва: " &lt;&lt; subj-&gt;title &lt;&lt; std::endl;
        for (int i = spec-&gt;studentCount - 1; i &gt;= 0; i--) {
            Student* s = spec-&gt;students[i];
            s-&gt;credits += subj-&gt;credits;
            std::cout &lt;&lt; "  " &lt;&lt; s-&gt;name &lt;&lt; " -> " &lt;&lt; s-&gt;credits &lt;&lt; " кредита" &lt;&lt; std::endl;
            checkGraduation(s);
        }
    }

    // Специалността изкарва предмет, произволни студенти са скъсани
    void passSubjectWithFails(Specialty* spec, Subject* subj, int failCount) {
        if (spec != subj-&gt;specialty) return;
        std::cout &lt;&lt; spec-&gt;name &lt;&lt; " изкарва " &lt;&lt; subj-&gt;title
                  &lt;&lt; " (" &lt;&lt; failCount &lt;&lt; " скъсани)" &lt;&lt; std::endl;
        for (int i = spec-&gt;studentCount - 1; i &gt;= 0; i--) {
            Student* s = spec-&gt;students[i];
            bool failed = (failCount &gt; 0) &amp;&amp; (rand() % spec-&gt;studentCount &lt; failCount);
            if (failed) {
                std::cout &lt;&lt; "  " &lt;&lt; s-&gt;name &lt;&lt; " -> СКЪСАН" &lt;&lt; std::endl;
                failCount--;
            } else {
                s-&gt;credits += subj-&gt;credits;
                std::cout &lt;&lt; "  " &lt;&lt; s-&gt;name &lt;&lt; " -> " &lt;&lt; s-&gt;credits &lt;&lt; " кредита" &lt;&lt; std::endl;
                checkGraduation(s);
            }
        }
    }

    // Прехвърля студент в друга специалност (нужни са &gt;= 30 кредита)
    void transfer(Student* s, Specialty* newSpec) {
        if (s-&gt;credits &lt; CREDITS_TRANSFER) {
            std::cout &lt;&lt; s-&gt;name &lt;&lt; " няма достатъчно кредити ("
                      &lt;&lt; s-&gt;credits &lt;&lt; "/30)." &lt;&lt; std::endl;
            return;
        }
        Specialty* old = s-&gt;specialty;
        old-&gt;removeStudent(s);
        newSpec-&gt;addStudent(s);
        std::cout &lt;&lt; s-&gt;name &lt;&lt; " се прехвърли от "
                  &lt;&lt; old-&gt;name &lt;&lt; " в " &lt;&lt; newSpec-&gt;name &lt;&lt; std::endl;
    }

    // Извежда съдържанието на университета
    void print() const {
        std::cout &lt;&lt; "\n=== Университет ===" &lt;&lt; std::endl;
        for (int i = 0; i &lt; specialtyCount; i++) {
            const Specialty&amp; sp = specialties[i];
            std::cout &lt;&lt; "Специалност: " &lt;&lt; sp.name
                      &lt;&lt; " (" &lt;&lt; sp.studentCount &lt;&lt; " студенти)" &lt;&lt; std::endl;
            for (int j = 0; j &lt; sp.studentCount; j++) {
                std::cout &lt;&lt; "  - " &lt;&lt; sp.students[j]-&gt;name
                          &lt;&lt; " | FN: " &lt;&lt; sp.students[j]-&gt;fn
                          &lt;&lt; " | Кредити: " &lt;&lt; sp.students[j]-&gt;credits
                          &lt;&lt; std::endl;
            }
        }
        std::cout &lt;&lt; "\nПредмети:" &lt;&lt; std::endl;
        for (int i = 0; i &lt; subjectCount; i++) {
            std::cout &lt;&lt; "  " &lt;&lt; subjects[i].title

                      &lt;&lt; " | " &lt;&lt; subjects[i].credits &lt;&lt; " кредита"
                      &lt;&lt; " | " &lt;&lt; subjects[i].specialty-&gt;name
                      &lt;&lt; std::endl;
        }
    }
};

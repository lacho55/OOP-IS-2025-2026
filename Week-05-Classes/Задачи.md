## Task 1

Да се напишат следните класове:
- **Учен** — всеки учен има ime, длъжност, заплата и колко пари печели за университета с труда си за един ден.
- **Университет** — в него има учени.

*Академичните длъжности са: **асистент, главен асистент, доцент, професор***

Напишете функции, които да позволяват следните действия:
- Да се назначи учен.
- Да се уволни учен.
- Учен да се повиши *(разликата в заплатите между всички длъжности е 100 лв.)*
- Да се намалят заплатите на всички учени на дадена длъжност.
- Да се съкратят всички учени на дадена длъжност.
- Да се сметне печалбата *(profit)* за един месец — приход *(income)* минус разходи *(expenses)*.
- Да се изведат на конзолата всички служители и печалбата.

<details>
<summary>Решение</summary>

```cpp
#include <iostream>
#include <cstring>

const int MAX_SCIENTISTS = 100;
const int WORK_DAYS      = 20;

const int ASSISTANT   = 0;
const int SENIOR_ASST = 1;
const int DOCENT      = 2;
const int PROFESSOR   = 3;

const char*  TITLES[]    = { "Асистент", "Главен асистент", "Доцент", "Професор" };
const double BASE_SALARY = 1000.0;
const double SALARY_STEP =  100.0;

struct Scientist {
    char   name[64];
    int    rank;
    double salary;
    double dailyIncome;
};

struct University {
    Scientist scientists[MAX_SCIENTISTS];
    int       count;

    University() : count(0) {}

    void hire(const char* name, int rank, double dailyIncome) {
        if (count >= MAX_SCIENTISTS) {
            std::cout << "Няма свободни места!" << std::endl;
            return;
        }
        strncpy(scientists[count].name, name, 63);
        scientists[count].rank        = rank;
        scientists[count].salary      = BASE_SALARY + rank * SALARY_STEP;
        scientists[count].dailyIncome = dailyIncome;
        count++;
        std::cout << name << " е назначен." << std::endl;
    }

    // Уволнява по индекс — запълва дупката с последния елемент
    void fire(int index) {
        if (index < 0 || index >= count) return;
        std::cout << scientists[index].name << " е уволнен." << std::endl;
        scientists[index] = scientists[count - 1];
        count--;
    }

    void fire(const char* name) {
        for (int i = 0; i < count; i++) {
            if (strcmp(scientists[i].name, name) == 0) { fire(i); return; }
        }
        std::cout << "Учен с това ime не е намерен." << std::endl;
    }

    void promote(const char* name) {
        for (int i = 0; i < count; i++) {
            if (strcmp(scientists[i].name, name) == 0) {
                if (scientists[i].rank == PROFESSOR) {
                    std::cout << scientists[i].name << " вече е Професор." << std::endl;
                    return;
                }
                scientists[i].rank++;
                scientists[i].salary += SALARY_STEP;
                std::cout << scientists[i].name << " е повишен до "
                          << TITLES[scientists[i].rank] << std::endl;
                return;
            }
        }
        std::cout << "Учен с това ime не е намерен." << std::endl;
    }

    void reduceSalaries(int rank, double amount) {
        for (int i = 0; i < count; i++) {
            if (scientists[i].rank == rank)
                scientists[i].salary -= amount;
        }
        std::cout << "Заплатите на " << TITLES[rank]
                  << " са намалени с " << amount << " лв." << std::endl;
    }

    void fireByRank(int rank) {
        for (int i = count - 1; i >= 0; i--) {
            if (scientists[i].rank == rank) fire(i);
        }
    }

    double monthlyProfit() const {
        double income = 0, expenses = 0;
        for (int i = 0; i < count; i++) {
            income   += scientists[i].dailyIncome * WORK_DAYS;
            expenses += scientists[i].salary;
        }
        return income - expenses;
    }

    void print() const {
        std::cout << "\n=== Служители ===" << std::endl;
        double income = 0, expenses = 0;
        for (int i = 0; i < count; i++) {
            std::cout << "  " << scientists[i].name
                      << " | " << TITLES[scientists[i].rank]
                      << " | Заплата: "       << scientists[i].salary
                      << " | Дневен приход: " << scientists[i].dailyIncome
                      << std::endl;
            income   += scientists[i].dailyIncome * WORK_DAYS;
            expenses += scientists[i].salary;
        }
        std::cout << "Приход:  " << income            << " лв.\n"
                  << "Разходи: " << expenses          << " лв.\n"
                  << "Печалба: " << (income-expenses) << " лв." << std::endl;
    }
};

int main() {
    University uni;
    uni.hire("Иван Иванов",   ASSISTANT,  80);
    uni.hire("Мария Петрова", DOCENT,    150);
    uni.hire("Георги Колев",  PROFESSOR, 200);
    uni.hire("Петър Стоянов", ASSISTANT,  75);

    uni.promote("Иван Иванов");
    uni.reduceSalaries(ASSISTANT, 50);
    uni.print();

    uni.fireByRank(SENIOR_ASST);
    uni.print();
    return 0;
}
```

> 💡 Длъжностите са числа 0–3, повишението е просто `rank++` и `salary += 100`. При уволнение запълваме дупката с последния елемент, за да не местим целия масив.

## Task 2

Да се напишат следните класове:
- **Вафла** — марка, грамаж, цена на едро, цена за краен клиент.
- **Магазин за вафли** — масив от вафли, максимален брой вафли, приходи до момента, разходи до момента.

Напишете функции, които да позволяват следните действия:
- Продава се вафла.
- Поръчват се вафли от склада на дадена марка.
- Показва колко вафли има от дадена марка или даден грамаж.
- Показва печалбата до момента.

<details>
<summary>Решение</summary>

```cpp
#include <iostream>
#include <cstring>

const int MAX_TYPES = 50;

struct Wafer {
    char   brand[64];
    int    grams;
    double wholesalePrice;
    double retailPrice;
    int    quantity;
};

struct WaferShop {
    Wafer  stock[MAX_TYPES];
    int    typeCount;
    double income;
    double expenses;

    WaferShop() : typeCount(0), income(0), expenses(0) {}

    int findByBrand(const char* brand) const {
        for (int i = 0; i < typeCount; i++) {
            if (strcmp(stock[i].brand, brand) == 0) return i;
        }
        return -1;
    }

    void order(const char* brand, int grams,
               double wholesale, double retail, int qty) {
        int idx = findByBrand(brand);
        if (idx == -1) {
            if (typeCount >= MAX_TYPES) {
                std::cout << "Складът е пълен!" << std::endl;
                return;
            }
            strncpy(stock[typeCount].brand, brand, 63);
            stock[typeCount].grams          = grams;
            stock[typeCount].wholesalePrice = wholesale;
            stock[typeCount].retailPrice    = retail;
            stock[typeCount].quantity       = qty;
            typeCount++;
        } else {
            stock[idx].quantity += qty;
        }
        expenses += wholesale * qty;
        std::cout << "Поръчани " << qty << " бр. " << brand << std::endl;
    }

    void sell(const char* brand) {
        int idx = findByBrand(brand);
        if (idx == -1 || stock[idx].quantity == 0) {
            std::cout << "Няма налични вафли от марка " << brand << std::endl;
            return;
        }
        stock[idx].quantity--;
        income += stock[idx].retailPrice;
        std::cout << "Продадена " << brand
                  << " за " << stock[idx].retailPrice << " лв." << std::endl;
    }

    void countByBrand(const char* brand) const {
        int idx = findByBrand(brand);
        if (idx == -1) {
            std::cout << "Марката " << brand << " не е намерена." << std::endl;
            return;
        }
        std::cout << "Марка " << brand << ": " << stock[idx].quantity << " бр." << std::endl;
    }

    void countByGrams(int grams) const {
        int total = 0;
        for (int i = 0; i < typeCount; i++) {
            if (stock[i].grams == grams) total += stock[i].quantity;
        }
        std::cout << "Вафли " << grams << "г: " << total << " бр." << std::endl;
    }

    void printProfit() const {
        std::cout << "Приходи:  " << income            << " лв.\n"
                  << "Разходи:  " << expenses          << " лв.\n"
                  << "Печалба:  " << (income-expenses) << " лв." << std::endl;
    }
};

int main() {
    WaferShop shop;
    shop.order("Kinder", 21, 0.80, 1.20, 50);
    shop.order("Balsen", 35, 0.60, 1.00, 30);
    shop.order("Kinder", 21, 0.80, 1.20, 20);

    shop.sell("Kinder");
    shop.sell("Balsen");
    shop.sell("Kinder");

    shop.countByBrand("Kinder");
    shop.countByGrams(21);
    shop.printProfit();
    return 0;
}
```

> 💡 При повторна поръчка на същата марка само увеличаваме `quantity`, без нов запис. Разходите се отчитат при поръчка, приходите — при продажба.

## Task 3

Да се напишат следните класове:
- **Студент** — имена, факултетен номер, специалност и общ брой получени кредити.
- **Специалност** — съдържа всички студенти, записани в нея.
- **Предмет** — заглавие, преподавател, специалност и брой кредити.
- **Университет** — специалности и предмети.

Напишете функции, които да позволяват следните действия:
- Студент да бъде приет в дадена специалност.
- Една специалност да "изкара" даден предмет.
- Една специалност да "изкара" даден предмет, като произволни студенти са "скъсани".
- Студент да може да премине от една в друга специалност, стига да има натрупани 30 кредита.
- Студент да завърши — автоматично след 180 кредита.
- Да се изведе на конзолата съдържанието на университета.

<details>
<summary>Решение</summary>

```cpp
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <ctime>

const int MAX_STUDENTS    = 200;
const int MAX_SPECIALTIES =  20;
const int MAX_SUBJECTS    =  50;
const int CREDITS_TRANSFER =  30;
const int CREDITS_GRADUATE = 180;

struct Specialty;

struct Student {
    char      name[64];
    int       fn;
    int       credits;
    bool      graduated;
    Specialty* specialty;

    Student() : fn(0), credits(0), graduated(false), specialty(nullptr) {
        name[0] = '\0';
    }
};

struct Specialty {
    char     name[64];
    Student* students[MAX_STUDENTS];
    int      studentCount;

    Specialty() : studentCount(0) { name[0] = '\0'; }

    bool addStudent(Student* s) {
        if (studentCount >= MAX_STUDENTS) return false;
        students[studentCount++] = s;
        s->specialty = this;
        return true;
    }

    void removeStudent(Student* s) {
        for (int i = 0; i < studentCount; i++) {
            if (students[i] == s) {
                students[i] = students[studentCount - 1];
                studentCount--;
                return;
            }
        }
    }
};

struct Subject {
    char       title[64];
    char       teacher[64];
    Specialty* specialty;
    int        credits;

    Subject() : specialty(nullptr), credits(0) {
        title[0] = teacher[0] = '\0';
    }
};

struct University {
    Student   students[MAX_STUDENTS];
    int       studentCount;

    Specialty specialties[MAX_SPECIALTIES];
    int       specialtyCount;

    Subject   subjects[MAX_SUBJECTS];
    int       subjectCount;

    University() : studentCount(0), specialtyCount(0), subjectCount(0) {}

    Specialty* addSpecialty(const char* name) {
        if (specialtyCount >= MAX_SPECIALTIES) return nullptr;
        strncpy(specialties[specialtyCount].name, name, 63);
        return &specialties[specialtyCount++];
    }

    Subject* addSubject(const char* title, const char* teacher,
                        Specialty* spec, int credits) {
        if (subjectCount >= MAX_SUBJECTS) return nullptr;
        strncpy(subjects[subjectCount].title,   title,   63);
        strncpy(subjects[subjectCount].teacher, teacher, 63);
        subjects[subjectCount].specialty = spec;
        subjects[subjectCount].credits   = credits;
        return &subjects[subjectCount++];
    }

    Student* enroll(const char* name, int fn, Specialty* spec) {
        if (studentCount >= MAX_STUDENTS) return nullptr;
        strncpy(students[studentCount].name, name, 63);
        students[studentCount].fn        = fn;
        students[studentCount].credits   = 0;
        students[studentCount].graduated = false;
        spec->addStudent(&students[studentCount]);
        std::cout << name << " е приет в " << spec->name << std::endl;
        return &students[studentCount++];
    }

    void checkGraduation(Student* s) {
        if (!s->graduated && s->credits >= CREDITS_GRADUATE) {
            s->graduated = true;
            s->specialty->removeStudent(s);
            std::cout << "  [" << s->name << " ЗАВЪРШИ!]" << std::endl;
        }
    }

    void passSubject(Specialty* spec, Subject* subj) {
        if (spec != subj->specialty) {
            std::cout << "Предметът не е за тази специалност!" << std::endl;
            return;
        }
        std::cout << spec->name << " изкарва: " << subj->title << std::endl;
        for (int i = spec->studentCount - 1; i >= 0; i--) {
            Student* s = spec->students[i];
            s->credits += subj->credits;
            std::cout << "  " << s->name << " -> " << s->credits << " кредита" << std::endl;
            checkGraduation(s);
        }
    }

    void passSubjectWithFails(Specialty* spec, Subject* subj, int failCount) {
        if (spec != subj->specialty) return;
        std::cout << spec->name << " изкарва " << subj->title
                  << " (" << failCount << " скъсани)" << std::endl;
        for (int i = spec->studentCount - 1; i >= 0; i--) {
            Student* s = spec->students[i];
            bool failed = (failCount > 0) && (rand() % spec->studentCount < failCount);
            if (failed) {
                std::cout << "  " << s->name << " -> СКЪСАН" << std::endl;
                failCount--;
            } else {
                s->credits += subj->credits;
                std::cout << "  " << s->name << " -> " << s->credits << " кредита" << std::endl;
                checkGraduation(s);
            }
        }
    }

    void transfer(Student* s, Specialty* newSpec) {
        if (s->credits < CREDITS_TRANSFER) {
            std::cout << s->name << " няма достатъчно кредити ("
                      << s->credits << "/30)." << std::endl;
            return;
        }
        Specialty* old = s->specialty;
        old->removeStudent(s);
        newSpec->addStudent(s);
        std::cout << s->name << " се прехвърли от "
                  << old->name << " в " << newSpec->name << std::endl;
    }

    void print() const {
        std::cout << "\n=== Университет ===" << std::endl;
        for (int i = 0; i < specialtyCount; i++) {
            const Specialty& sp = specialties[i];
            std::cout << "Специалност: " << sp.name
                      << " (" << sp.studentCount << " студенти)" << std::endl;
            for (int j = 0; j < sp.studentCount; j++) {
                std::cout << "  - " << sp.students[j]->name
                          << " | FN: "      << sp.students[j]->fn
                          << " | Кредити: " << sp.students[j]->credits
                          << std::endl;
            }
        }
        std::cout << "\nПредмети:" << std::endl;
        for (int i = 0; i < subjectCount; i++) {
            std::cout << "  " << subjects[i].title
                      << " | " << subjects[i].teacher
                      << " | " << subjects[i].credits << " кредита"
                      << " | " << subjects[i].specialty->name
                      << std::endl;
        }
    }
};

int main() {
    srand((unsigned)time(nullptr));

    University uni;
    Specialty* cs  = uni.addSpecialty("Информатика");
    Specialty* mat = uni.addSpecialty("Математика");

    Subject* oop  = uni.addSubject("ООП",       "Проф. Петров", cs,  10);
    Subject* algo = uni.addSubject("Алгоритми", "Доц. Иванова", cs,   8);
    Subject* ana  = uni.addSubject("Анализ",    "Проф. Колев",  mat, 10);

    Student* ivan  = uni.enroll("Иван Иванов",   70001, cs);
    Student* maria = uni.enroll("Мария Петрова", 70002, cs);
    Student* peter = uni.enroll("Петър Стоянов", 70003, mat);

    uni.passSubject(cs, oop);
    uni.passSubjectWithFails(cs, algo, 1);
    uni.transfer(ivan, mat);
    uni.passSubject(mat, ana);

    uni.print();
    return 0;
}
```

> 💡 Студентите живеят в масива на `University`, а `Specialty` пази само **указатели** към тях — без копиране. При завършване студентът се премахва от специалността, но остава в университетския масив с флаг `graduated = true`.

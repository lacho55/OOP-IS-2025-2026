Задача 3 — Обръщане на масив

Функция reverse, която обръща масив на място.
Solution 
// Task 3
#include <iostream>
using namespace std;

void reverse(int* arr, int n) {
    int left = 0, right = n - 1;
    while (left < right) {
        int temp = arr[left];
        arr[left]  = arr[right];
        arr[right] = temp;
        left++;
        right--;
    }
}

int main() {
    int a[] = {1, 2, 3, 4, 5};
    reverse(a, 5);
    for (int i = 0; i < 5; i++)
        cout << a[i] << " ";
    cout << endl; // 5 4 3 2 1
    return 0;
}

Задача 4 — Сливане на два сортирани масива

Слива два сортирани масива в трети, така че резултатът също да е сортиран (без допълнително сортиране).
Solution 
// Task 4
#include <iostream>
using namespace std;

// Precondition: result must have size n1+n2
void mergeSorted(int* a, int n1, int* b, int n2, int* result) {
    int i = 0, j = 0, k = 0;
    while (i < n1 && j < n2) {
        if (a[i] <= b[j]) result[k++] = a[i++];
        else               result[k++] = b[j++];
    }
    while (i < n1) result[k++] = a[i++];
    while (j < n2) result[k++] = b[j++];
}

int main() {
    int a[] = {1, 3, 5};
    int b[] = {2, 4, 6, 8};
    int n1 = 3, n2 = 4;
    int* merged = new int[n1 + n2];

    mergeSorted(a, n1, b, n2, merged);
    for (int i = 0; i < n1 + n2; i++)
        cout << merged[i] << " ";
    cout << endl; // 1 2 3 4 5 6 8

    delete[] merged;
    return 0;
}

Задача 5 — Последен елемент равен на средното на съседите

Връща индекса на последния елемент, равен на средното аритметично на съседите си. Крайните елементи имат само един съсед. Връща -1 при липса.
Solution 
// Task 5
#include <iostream>
using namespace std;

int lastAvgNeighbor(double* arr, int n) {
    int result = -1;
    for (int i = 0; i < n; i++) {
        double avg;
        if (i == 0)         avg = arr[1];
        else if (i == n-1)   avg = arr[n-2];
        else                avg = (arr[i-1] + arr[i+1]) / 2.0;

        if (arr[i] == avg) result = i;
    }
    return result;
}

int main() {
    double arr[] = {1, 3, 2, 4, 3};
    // arr[2]=2, avg(1,3)=2 ✓  arr[4]=3, avg(4)=4 ✗  arr[2] is result
    cout << lastAvgNeighbor(arr, 5) << endl; // 2
    return 0;
}

Задача 6 — Премахване на всички срещания на число

Премахва всички срещания на дадено число от масив (на място, намалява размера). Връща true ако е премахнато поне едно срещане.
Solution 
// Task 6
#include <iostream>
using namespace std;

bool removeAll(int* arr, int& n, int val) {
    int writeIdx = 0;
    bool found = false;
    for (int i = 0; i < n; i++) {
        if (arr[i] == val) {
            found = true;
        } else {
            arr[writeIdx++] = arr[i];
        }
    }
    n = writeIdx;
    return found;
}

int main() {
    int arr[] = {1, 3, 2, 3, 4, 3};
    int n = 6;
    bool removed = removeAll(arr, n, 3);
    cout << (removed ? "Removed" : "Not found") << endl;
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << endl; // 1 2 4
    return 0;
}

Задача 7 — Двумерен динамичен масив

Създава 2D масив с произволни числа [-50, 50], функция за извеждане, sumByRow и sumByCol, и освобождаване на паметта.
Solution 
// Task 7
#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

void printMatrix(int** m, int rows, int cols) {
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++)
            cout << m[i][j] << "\t";
        cout << endl;
    }
}

int* sumByRow(int** m, int rows, int cols) {
    int* sums = new int[rows]();
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            sums[i] += m[i][j];
    return sums;
}

int* sumByCol(int** m, int rows, int cols) {
    int* sums = new int[cols]();
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            sums[j] += m[i][j];
    return sums;
}

int main() {
    srand(time(0));
    int rows, cols;
    cout << "Rows: "; cin >> rows;
    cout << "Cols: "; cin >> cols;

    int** m = new int*[rows];
    for (int i = 0; i < rows; i++) {
        m[i] = new int[cols];
        for (int j = 0; j < cols; j++)
            m[i][j] = (rand() % 101) - 50;
    }

    printMatrix(m, rows, cols);

    int* rowSums = sumByRow(m, rows, cols);
    int* colSums = sumByCol(m, rows, cols);

    cout << "Row sums: ";
    for (int i = 0; i < rows; i++) cout << rowSums[i] << " ";
    cout << endl;

    cout << "Col sums: ";
    for (int j = 0; j < cols; j++) cout << colSums[j] << " ";
    cout << endl;

    delete[] rowSums;
    delete[] colSums;
    for (int i = 0; i < rows; i++)
        delete[] m[i];
    delete[] m;
    return 0;
}

Задача 8 — Функции за управление на масив

createArray, initArray, createInitArray, printArray, freeArray — с повторна инициализация на нов масив на същия адрес.
Solution 
// Task 8
#include <iostream>
using namespace std;

int* createArray(int size) {
    return new int[size];
}

void initArray(int* arr, int size) {
    cout << "Въведете " << size << " числа: ";
    for (int i = 0; i < size; i++)
        cin >> arr[i];
}

int* createInitArray(int size) {
    int* arr = createArray(size);
    initArray(arr, size);
    return arr;
}

void printArray(int* arr, int size) {
    for (int i = 0; i < size; i++)
        cout << arr[i] << " ";
    cout << endl;
}

void freeArray(int* arr) {
    delete[] arr;
}

int main() {
    int size1;
    cout << "Размер на първия масив: ";
    cin >> size1;
    int* arr = createInitArray(size1);
    printArray(arr, size1);
    freeArray(arr);

    // Re-initialize at same pointer variable with new size
    int size2;
    cout << "Размер на втория масив: ";
    cin >> size2;
    arr = createInitArray(size2);
    printArray(arr, size2);
    freeArray(arr);

    return 0;
}

⭐ BONUS — Двоично търсене с указател

Функция, която търси число в масив (двоично ако е сортиран) и връща указател към първото срещане, или NULL.
Solution 
// BONUS — Binary search returning pointer
#include <iostream>
using namespace std;

// Precondition: array must be sorted for binary search to work correctly
int* binarySearch(int* arr, int n, int target) {
    int left = 0, right = n - 1;
    int* found = nullptr;

    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target) {
            found = &arr[mid];
            right = mid - 1; // keep searching left for FIRST occurrence
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return found; // nullptr if not found
}

int main() {
    int arr[] = {1, 2, 3, 3, 5, 8, 13};
    int n = 7;

    int* ptr = binarySearch(arr, n, 3);
    if (ptr) cout << "Found: " << *ptr << " at index " << (ptr - arr) << endl;
    else     cout << "Not found" << endl;

    ptr = binarySearch(arr, n, 7);
    if (ptr) cout << "Found: " << *ptr << endl;
    else     cout << "Not found" << endl;

    return 0;
}
